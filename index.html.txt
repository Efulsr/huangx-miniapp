        function getRandomSymbol() {
            const rand = Math.random();
            if (rand < 0.05) return '👑';
            if (rand < 0.15) return '🐋';
            if (rand < 0.35) return '🪙';
            if (rand < 0.55) return '🐡';
            if (rand < 0.75) return '🐠';
            return '🐟';

        function calculateReward(results) {
            let total = results.reduce((sum, s) => sum + (rewards[s] || 5), 0);
            if (userData.boosters.includes('2x_poin')) total *= 2;
            return total;
        }

        // ==================== LEADERBOARD ====================
        function updateLeaderboard() {
            let allPlayers = JSON.parse(localStorage.getItem('huangx_leaderboard') || '[]');
            const existingIndex = allPlayers.findIndex(p => p.userId === userData.userId);
            const playerData = {
                userId: userData.userId,
                name: userData.userName,
                coins: userData.coins,
                lastUpdate: Date.now()
            };
            
            if (existingIndex !== -1) allPlayers[existingIndex] = playerData;
            else allPlayers.push(playerData);
            
            allPlayers.sort((a, b) => b.coins - a.coins);
            localStorage.setItem('huangx_leaderboard', JSON.stringify(allPlayers.slice(0, 100)));
            
            if (document.getElementById('leaderboardModal').style.display === 'flex') {
                loadLeaderboardTab('daily');
            }
        }

        function loadLeaderboardTab(tabType) {
            const allPlayers = JSON.parse(localStorage.getItem('huangx_leaderboard') || '[]');
            const contentDiv = document.getElementById('leaderboardContent');
            
            if (allPlayers.length === 0) {
                contentDiv.innerHTML = '<div class="loading">Belum ada pemain. Ajak teman main!</div>';
                return;
            }
            
            let filteredPlayers = [...allPlayers];
            if (tabType === 'daily') {
                const today = new Date().toDateString();
                filteredPlayers = allPlayers.filter(p => new Date(p.lastUpdate).toDateString() === today);
                if (filteredPlayers.length === 0) filteredPlayers = allPlayers.slice(0, 10);
            }
            if (tabType === 'weekly') {
                const weekAgo = Date.now() - 7 * 24 * 60 * 60 * 1000;
                filteredPlayers = allPlayers.filter(p => p.lastUpdate > weekAgo);
                if (filteredPlayers.length === 0) filteredPlayers = allPlayers.slice(0, 10);
            }
            
            filteredPlayers.sort((a, b) => b.coins - a.coins);
            const topPlayers = filteredPlayers.slice(0, 10);
            
            let html = '<ul class="leaderboard-list">';
            topPlayers.forEach((player, idx) => {
                const rank = idx + 1;
                let rankClass = rank === 1 ? 'rank-1' : rank === 2 ? 'rank-2' : rank === 3 ? 'rank-3' : '';
                html += `
                    <li class="leaderboard-item ${rankClass}">
                        <div class="rank-number">${rank === 1 ? '👑' : rank === 2 ? '🥈' : rank === 3 ? '🥉' : `#${rank}`}</div>
                        <div class="rank-name">${escapeHtml(player.name)}</div>
                        <div class="rank-coins">${player.coins.toLocaleString()} 🪙</div>
                    </li>
                `;
            });
            html += '</ul>';
            if (topPlayers.length === 0) html = '<div class="loading">Belum ada data</div>';
            contentDiv.innerHTML = html;
            updateMyRank();
        }

        function updateMyRank() {

            document.getElementById('myRank').innerHTML = `📊 Peringkat Anda: ${rankText} | Total: ${userData.coins.toLocaleString()} 🪙`;
        }

        function escapeHtml(text) {

        // ==================== HADIAH ====================
        function getWeekNumber(date) {
            const d = new Date(date);
            d.setHours(0, 0, 0, 0);
            d.setDate(d.getDate() + 3 - (d.getDay() + 6) % 7);
            const week1 = new Date(d.getFullYear(), 0, 4);
            return 1 + Math.round(((d - week1) / 86400000 - 3 + (week1.getDay() + 6) % 7) / 7);
        }

        function loadRewardClaims() {
            const saved = localStorage.getItem('huangx_reward_claims');
            if (saved) rewardClaims = JSON.parse(saved);
            if (!rewardClaims.weekly) rewardClaims.weekly = {};
            if (!rewardClaims.lastResetDate) rewardClaims.lastResetDate = null;
        }

        }

        async function claimReward() {
            if (!userData.userId) {
                showMessage('❌ Silakan buka game dari Telegram terlebih dahulu!');
                return;
            }
            
            const currentWeekId = getWeekNumber(new Date());
            const claimKey = `${currentWeekId}_${userData.userId}`;
            const myReward = rewardClaims.weekly[claimKey];
            
            if (!myReward) {
                showMessage('❌ Kamu tidak memenangkan hadiah minggu ini. Coba lagi minggu depan!');
                return;
            }
            
            if (myReward.claimed) {
                showMessage('✅ Hadiah sudah pernah diklaim!');
                return;
            }
            
            showMessage(`🎉 Selamat! Anda mendapat ${myReward.amount} SOL! Hubungi @HuangXBot untuk klaim.`);
            myReward.claimed = true;
            saveRewardClaims();
        }
